Vraag 1a - @supports (display: grid)

@supports (display: grid) {
  div {
    background: blue;
  }
}
div {
  background: red;
}

//@support regel wordt uitgevoerd als de browser grid ondersteunt
//door de cascade wordt de div toch rood
